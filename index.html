function render(data) { const now = new Date(); const timeStr = now.getHours().toString().padStart(2, '0') + ":" + now.getMinutes().toString().padStart(2, '0'); const upcoming = data.filter(b => b.time >= timeStr).slice(0, 4); const container = document.getElementById('nextBusContainer'); // 乗り場判定（長野駅のときだけ有効にするための判定も含む） const getPlatformBadge = (line) => { // 現在のバス停名を取得 const stopName = document.getElementById('currentStopLabel').innerText; // 「長野駅」という文字が含まれていない場合は何も出さない if (!stopName.includes("長野駅")) return ""; const lineStr = String(line); let num = ""; if (["10", "臨10", "11", "16", "17"].includes(lineStr)) num = "①"; else if (lineStr.startsWith("3")) num = "③"; else if (lineStr.startsWith("7")) num = "⑦"; if (!num) return ""; // デザイン：白背景に紺文字のバッジ return `長野駅${num}`; }; if (upcoming.length === 0) { container.innerHTML = '

運行終了

'; } else { container.innerHTML = upcoming.map((b, i) => { const isFirst = i === 0; const platBadge = getPlatformBadge(b.lineNum); const isExp = b.lineNum === "31快"; const ords = ["1ST","2ND","3RD","4TH"]; let subNote = ""; if (isExp) { subNote = currentRoute === 'to_zenkoji' ? `
（長野駅まで止まりません）
` : `
（丹波島橋南まで止まりません）
`; } return `
${platBadge} ${b.lineNum} ${ords[i]}

${b.destination}

${subNote}
${b.time}
`; }).join(''); } // フルリスト側 document.getElementById('fullListContent').innerHTML = data.map(b => { const platBadge = getPlatformBadge(b.lineNum); const isExp = b.lineNum === "31快"; let listNote = ""; if (isExp) { listNote = currentRoute === 'to_zenkoji' ? '
（長野駅まで止まりません）
' : '
（丹波島橋南まで止まりません）
'; } return `
${platBadge} ${b.lineNum} ${b.destination} ${b.time}
${listNote}
`; }).join(''); }
